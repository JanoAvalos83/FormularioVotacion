-- Crear tabla region
CREATE TABLE region (
    id_region INT PRIMARY KEY,
    nombre_region VARCHAR(255)
);

-- Crear tabla comoseentero
CREATE TABLE comoseentero (
    id_como_entero INT PRIMARY KEY,
    nombre_entero VARCHAR(255)
);

-- Crear tabla comuna
CREATE TABLE comuna (
    id_comuna INT PRIMARY KEY,
    nombre_comuna VARCHAR(255),
    id_region INT,
    FOREIGN KEY (id_region) REFERENCES region(id_region)
);

-- Crear tabla candidato
CREATE TABLE candidato (
    id_candidato INT PRIMARY KEY,
    nombre_candidato VARCHAR(255),
    email VARCHAR(255),
    votos INT
);
-- Crear tabla persona
CREATE TABLE persona (
    id_persona INT PRIMARY KEY,
    nombre_apellido VARCHAR(255),
    alias VARCHAR(255),
    rut VARCHAR(20),
    email VARCHAR(255),
    id_comuna INT,
    id_candidato INT,
    FOREIGN KEY (id_comuna) REFERENCES comuna(id_comuna),
    FOREIGN KEY (id_candidato) REFERENCES candidato(id_candidato)
);


-- CrearTabla personaComoentero

CREATE TABLE personacomoentero (
    id_personaconocio INT PRIMARY KEY,
    id_como_entero INT,
    id_persona INT,
    FOREIGN KEY (id_como_entero) REFERENCES comoseentero(id_como_entero),
    FOREIGN KEY (id_persona) REFERENCES persona(id_persona)
);

    
-- crear tablas para secuencias  y TRIGGER correspondientes
 
-- Creartabla secuencias
CREATE TABLE secuencia (
  nombre_tabla VARCHAR(50) NOT NULL,
  nombre_secuencia VARCHAR(50) NOT NULL,
  ultimo_valor INT(11) NOT NULL DEFAULT 0,
  PRIMARY KEY (nombre_tabla, nombre_secuencia)
);

-- Insertar variable inicial tabla secuencias
INSERT INTO secuencia (nombre_tabla, nombre_secuencia, ultimo_valor)
VALUES ('persona', 'sq_persona', 0);

-- Insertar variable inicial tabla secuencias
INSERT INTO secuencia (nombre_tabla, nombre_secuencia, ultimo_valor)
VALUES ('personacomoentero', 'sq_personacomoentero', 0);

DELIMITER //
CREATE TRIGGER tr_incrementar_secuencia
BEFORE INSERT ON persona
FOR EACH ROW
BEGIN
  UPDATE secuencia
  SET ultimo_valor = ultimo_valor + 1
  WHERE nombre_tabla = 'persona' AND nombre_secuencia = 'sq_persona';

  SET NEW.id_persona = (SELECT ultimo_valor FROM secuencia
                WHERE nombre_tabla = 'persona' AND nombre_secuencia = 'sq_persona');
END //
DELIMITER ;

DELIMITER //
CREATE TRIGGER tr_incrementar_secuencia_personacomoentero
BEFORE INSERT ON personacomoentero
FOR EACH ROW
BEGIN
  UPDATE secuencia
  SET ultimo_valor = ultimo_valor + 1
  WHERE nombre_tabla = 'personacomoentero' AND nombre_secuencia = 'sq_personacomoentero';

  SET NEW.id_personaconocio = (SELECT ultimo_valor FROM secuencia
                WHERE nombre_tabla = 'personacomoentero' AND nombre_secuencia = 'sq_personacomoentero');
END //
DELIMITER ;


DELIMITER //
CREATE TRIGGER after_insert_persona
AFTER INSERT ON persona
FOR EACH ROW
BEGIN
    DECLARE candidato_id INT;
    DECLARE votos_actuales INT;

    -- Obtener el ID del candidato de la persona insertada
    SELECT id_candidato INTO candidato_id FROM persona WHERE id_persona = NEW.id_persona;

    -- Incrementar el número de votos del candidato
    UPDATE candidato SET votos = votos + 1 WHERE id_candidato = candidato_id;
END //
DELIMITER ;
